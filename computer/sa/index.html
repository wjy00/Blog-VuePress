  // æ—¶é—´å¤æ‚åº¦æœ€åæƒ…å†µä¸‹ä¸ºO(N^2),é€šå¸¸æƒ…å†µä¸‹éƒ½å¥½äºO(N^2)
  shellSort() {
    let length = this.array.length;
    let gap = Math.floor(length / 2); // åˆå§‹åŒ–å¢é‡,é—´éš”é‡
    for (gap; gap > 0; gap = Math.floor(gap / 2)) {
      // ä»¥gapä½œä¸ºé—´éš”ï¼Œè¿›è¡Œåˆ†ç»„ï¼Œå¯¹åˆ†ç»„è¿›è¡Œæ’å…¥æ’åº
      for (let i = gap; i < length; i++) {
        for (let j = i - gap; j >= 0; j -= gap) {
          // è¿›è¡Œæ’å…¥æ’åº
          if (this.array[j] > this.array[j + gap]) {
            [this.array[j], this.array[j + gap]] = [this.array[j + gap], this.array[j]]
          }
        }
      }
    }
  }
}
